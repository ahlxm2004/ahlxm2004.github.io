众所周知， $$\int_{0}^{\pi/2}\sin^{2n}x\mathrm{d}x=\dfrac{\left(2n-1\right)!!}{\left(2n\right)!!}\cdot\dfrac{\pi}{2}$$ $$\int_{0}^{\pi/2}\sin^{2n+1}x\mathrm{d}x=\dfrac{\left(2n\right)!!}{\left(2n+1\right)!!}$$ 由 $\sin^{2n+1}x\lt\sin^{2n}x\lt\sin^{2n-1}x$ 积分化简得 $$\dfrac{2n}{2n+1}\cdot\dfrac{\pi}{2}\lt\dfrac{\left(\left(2n\right)!!\right)^2}{\left(2n+1\right)\left(\left(2n-1\right)!!\right)^2}\lt\dfrac{\pi}{2}$$ 两侧取极限，由夹逼定理可知 $$\sqrt{\pi}=\lim\limits_{n\to +\infty}\dfrac{\left(n!\right)^22^{2n}}{\left(2n\right)!\sqrt{n}}$$ 设 $\forall n\in\mathbb{N}^+,a_n=\dfrac{n!\mathrm{e}^n}{n^{n+\frac{1}{2}}}$，则 $\sqrt{\pi}=\lim\limits_{n\to +\infty}\dfrac{a_n^2}{\sqrt{2}a_{2n}}$
由 $\ln\dfrac{a_n}{a_{n+1}}=\left(n+\dfrac{1}{2}\right)\ln\left(1+\dfrac{1}{n}\right)-1\gt 0$ 可知 $a_n$ 严格递减，设 $\lim\limits_{n\to +\infty}a_n=\alpha$，则 $\alpha=\sqrt{2\pi}$。
此时已经可以得到一个初步的结论（即斯特林公式）： $$n!\sim\sqrt{2\pi n}\left(\dfrac{n}{\mathrm{e}}\right)^n$$ 换个话题，$\forall n\in\mathbb{N}^+,k\in\mathbb{N}^+,\sum\limits_{i=n}^{+\infty}\dfrac{1}{i^k}$ 的值是否可以用 $\dfrac{1}{n}$ 的幂级数逼近？
$k=1$ 时 $\sum\limits_{i=n}^{+\infty}\dfrac{1}{i^k}$ 发散，不需要考虑。
$k\gt 1$ 时由积分近似， $$\sum\limits_{i=n}^{+\infty}\dfrac{1}{i^k}=\dfrac{1}{\left(k-1\right)n^{k-1}}+O\left(\dfrac{1}{n^k}\right)$$ 设 $m\in\mathbb{N}^+$，存在一组 $A_i$ 使得 $$\sum\limits_{i=n}^{+\infty}\dfrac{1}{i^k}=\sum\limits_{i=k-1}^m\dfrac{A_i}{n^i}+O\left(\dfrac{1}{n^{m+1}}\right)$$ 则 $\sum\limits_{i=n}^{+\infty}\dfrac{1}{i^k}-\sum\limits_{i=k-1}^m\dfrac{A_i}{n^i}$ 是关于 $\dfrac{1}{n}$ 的多项式，由 Stolz 定理 $$\lim\limits_{n\to\infty}\dfrac{\sum\limits_{i=n}^{+\infty}\dfrac{1}{i^k}-\sum\limits_{i=k-1}^m\dfrac{A_i}{n^i}}{\dfrac{1}{n^m}}=\lim\limits_{n\to\infty}\dfrac{-\dfrac{1}{n^k}+\sum\limits_{i=k-1}^mA_i\dfrac{\left(n+1\right)^i-n^i}{n^i\left(n+1\right)^i}}{-\dfrac{m}{n^{m+1}}}=0$$ 因为分子是关于 $\dfrac{1}{n}$ 的多项式，所以 $A_{m+1}$ 存在，且 $$ \begin{aligned} A_{m+1} &= \lim\limits_{n\to\infty}\dfrac{-\dfrac{1}{n^k}+\sum\limits_{i=k-1}^mA_i\dfrac{\left(n+1\right)^i-n^i}{n^i\left(n+1\right)^i}}{-\dfrac{m+1}{n^{m+2}}}\\ &= \dfrac{1}{m+1}\left[\binom{m+1}{k-1}-\sum\limits_{i=k-1}^m A_i\binom{m+1}{i-1}\right] \end{aligned} $$ 一些例子： $$ \begin{aligned} \sum\limits_{i=n}^{+\infty}\dfrac{1}{i^2}&=\dfrac{1}{n}+\dfrac{1}{2n^2}+\dfrac{1}{6n^3}-\dfrac{1}{30n^5}+\dfrac{1}{42n^7}-\dfrac{1}{30n^9}+\omicron\left(\dfrac{1}{n^{10}}\right)\\ \sum\limits_{i=n}^{+\infty}\dfrac{1}{i^3}&=\dfrac{1}{2n^2}+\dfrac{1}{2n^3}+\dfrac{1}{4n^4}-\dfrac{1}{12n^6}+\dfrac{1}{12n^8}-\dfrac{3}{20n^{10}}+\omicron\left(\dfrac{1}{n^{10}}\right)\\ \sum\limits_{i=n}^{+\infty}\dfrac{1}{i^4}&=\dfrac{1}{3n^3}+\dfrac{1}{2n^4}+\dfrac{1}{3n^5}-\dfrac{1}{6n^7}+\dfrac{2}{9n^9}+\omicron\left(\dfrac{1}{n^{10}}\right) \end{aligned} $$ 设 $B_n^+$ 表示第二伯努利数，指数生成函数为 $\dfrac{x\mathrm{e}^x}{e^x-1}$；$B_n^-$ 表示第一伯努利数，指数生成函数为 $\dfrac{x}{e^x-1}$
由递推式容易推出 $$A_n=\dfrac{\dbinom{n}{k-1}}{n}B_{n-k+1}^+$$ 由 $$\dfrac{a_n}{a_{N+1}}=\exp\left[\sum\limits_{i=n}^N\left(i+\dfrac{1}{2}\right)\ln\left(1+\dfrac{1}{n}\right)-1\right]$$ 可知 $$ \begin{aligned} a_n &= \sqrt{2\pi}\exp\left[\sum\limits_{i=n}^{+\infty}\left(i+\dfrac{1}{2}\right)\ln\left(1+\dfrac{1}{n}\right)-1\right]\\ &= \sqrt{2\pi}\exp\left[\sum\limits_{i=1}^{+\infty}\left(i-1\right)!\left[x^{i-1}\right]\left(\dfrac{x-2}{2x^2}+\dfrac{1}{x\left(\mathrm{e}^x-1\right)}\right)\left(\dfrac{1}{n}\right)^i\right]\\ &= \sqrt{2\pi}\exp\left(\sum\limits_{i=1}^{+\infty}\dfrac{i\left(i+1\right)B_{i+1}^-}{n^i}\right)\\ \end{aligned} $$ 可以得到 $\ln\left(n!\right)$ 的斯特林级数展开 $$\ln\left(n!\right) = n\ln n-n+\dfrac{\ln n}{2}+\dfrac{\ln\left(2\pi\right)}{2}+\dfrac{1}{12n}-\dfrac{1}{360n^3}+\dfrac{1}{1260n^5}-\dfrac{1}{1680n^7}+\omicron\left(\dfrac{1}{n^8}\right)$$ 和 $n!$ 的斯特林级数展开 $$n! = \sqrt{2\pi n}\left(\dfrac{n}{\mathrm{e}}\right)^n\left(1+\dfrac{1}{12n}+\dfrac{1}{288n^2}-\dfrac{139}{51840n^3}-\dfrac{571}{2488320n^4}+\omicron\left(\dfrac{1}{n^4}\right)\right)$$ 需要注意的是，斯特林级数是渐进级数，收敛半径是 $0$。对于固定的 $n$，如果取的项数过多，误差反而会增大。
附 c++ 计算斯特林级数程序（__int128 范围内可计算至第 $18$ 项）： 
#include <bits/stdc++.h>

#define pc putchar

typedef __int128 ll; 

const ll L = 18;

void write_ll(ll x) {
	if (x < 0) pc('-'), x = -x;
	if (x > 9) write_ll(x / 10);
	pc(x % 10 + '0');
}

ll abs(ll x) {return x < 0 ? -x : x;}
ll gcd(ll x, ll y) {return y ? gcd(y, x % y) : x;}

struct num {
	ll x, y;
	num reduce() {ll d = gcd(abs(x), abs(y)); x /= d, y /= d; return *this;}
	num(ll _x = 0, ll _y = 1) {x = _x, y = _y; if (y < 0) x = -x, y = -y; reduce();}
	num &operator += (const num &t) {ll d = gcd(y, t.y); return *this = num(x * (t.y / d) + t.x * (y / d), y / d * t.y);}
	num &operator -= (const num &t) {ll d = gcd(y, t.y); return *this = num(x * (t.y / d) - t.x * (y / d), y / d * t.y);}
	num &operator *= (const num &t) {return *this = num(x * t.x, y * t.y);}
	num &operator /= (const num &t) {return *this = num(x * t.y, y * t.x);}
	num operator - () {return num(-x, y);}
};

num operator + (num A, num B) {return A += B;}
num operator - (num A, num B) {return A -= B;}
num operator * (num A, num B) {return A *= B;}
num operator / (num A, num B) {return A /= B;}

typedef std::vector <num> vector;

vector &operator += (vector &A, vector B) {
	for (ll i = 0; i < L; i++) A[i] += B[i];
	return A;
}

vector &operator *= (vector &A, vector B) {
	vector C(L);
	for (ll i = 0; i < L; i++)
		for (ll j = 0; i + j < L; j++)
			C[i + j] += A[i] * B[j];
	return A = C;
}

vector operator + (vector A, vector B) {return A += B;}
vector operator * (vector A, vector B) {return A *= B;}

vector Exp(vector A) {
	vector ans(L), tmp(L);
	ans[0] = tmp[0] = 1;
	for (ll i = 1; i < L; i++) {
		tmp *= A;
		for (ll j = 0; j < L; j++) tmp[j] /= i;
		ans += tmp;
	}
	return ans;
}

ll binom(ll x, ll y) {
	if (x < y || y < 0) return 0;
	ll ans = 1;
	for (ll i = x; i > x - y; i--) ans *= i;
	for (ll i = y; i; i--) ans /= i;
	return ans;
}

vector seris(ll n) {
	vector ans(L); ans[n - 1] = num(1, n - 1);
	for (ll i = n; i < L; i++) {
		ans[i] = binom(i, (i - 1) - (i - n));
		for (ll j = n - 1; j < i; j++)
			ans[i] -= ans[j] * binom(i, j - 1);
		ans[i] /= i;
	} 
	return ans;
}

int main() {
	vector ans(L);
	for (ll i = 1; i <= L; i++) {
		num cur = num(1, i + 1) - num(1, 2 * i);
		if (i & 1) cur = -cur;
		if (cur.x == 0) continue;
		vector tmp = seris(i);
		for (ll j = 0; j < L; j++) ans[j] += cur * tmp[j];
	}
	ans = Exp(ans);
	for (ll i = 0; i < L; i++)
		write_ll(ans[i].x), pc(' '), pc('/'),
		pc(' '), write_ll(ans[i].y), pc('\n');
	return 0;
}
